const auth = firebase.auth(); const db = firebase.database(); const storage = firebase.storage(); const directoryList = document.getElementById('directoryList'); const joinBtn = document.getElementById('joinBtn'); const joinModal = document.getElementById('joinModal'); const submitJoin = document.getElementById('submitJoin'); const photoInput = document.getElementById('photoInput'); const nameInput = document.getElementById('nameInput'); const roleInput = document.getElementById('roleInput'); // 🔍 Search filter document.getElementById('searchInput').addEventListener('input', e => { const q = e.target.value.toLowerCase(); const cards = directoryList.getElementsByClassName('directory-card'); for (let c of cards) { const name = c.querySelector('.name').textContent.toLowerCase(); c.style.display = name.includes(q) ? '' : 'none'; } }); // 🧩 Join button → Sign in via Google joinBtn.onclick = async () => { const provider = new firebase.auth.GoogleAuthProvider(); try { const result = await auth.signInWithPopup(provider); const user = result.user; joinModal.style.display = 'flex'; } catch (err) { alert('Sign-in failed.'); console.error(err); } }; joinModal.onclick = e => { if (e.target === joinModal) joinModal.style.display = 'none'; }; // 📤 Submit Join submitJoin.onclick = async () => { const user = auth.currentUser; if (!user) return alert('Please sign in first.'); const name = nameInput.value.trim(); const role = roleInput.value.trim(); const file = photoInput.files[0]; if (!name || !role || !file) return alert('Please fill in all fields and upload a photo.'); try { const fileRef = storage.ref('directory_photos/' + user.uid + '_' + file.name); await fileRef.put(file); const photoURL = await fileRef.getDownloadURL(); await db.ref('directory/' + user.uid).set({ name, email: user.email, role, photoURL, status: "Active", timestamp: Date.now() }); alert('Profile added successfully!'); joinModal.style.display = 'none'; nameInput.value = ''; roleInput.value = ''; photoInput.value = ''; } catch (err) { console.error(err); alert('Upload failed.'); } }; // 🔁 Load directory in real-time db.ref('directory').on('value', snapshot => { const data = snapshot.val() || {}; directoryList.innerHTML = ''; Object.values(data).forEach(m => { directoryList.innerHTML += `
${m.name}
${m.role}
🟢 ${m.status}
`; }); });
